131 static std::vector<std::string> split(const std::string& str, const std::string& delimiter = "=")
132 {
133  std::vector<std::string> _ret;
134  if(0 == delimiter.length() || std::string::npos == str.find(delimiter)) {
135  return _ret;
136  }
137 
138  std::size_t _find_offset = 0, _split_offset = 0;
139  while (std::string::npos != (_split_offset = str.find(delimiter, _find_offset))) {
140  _ret.push_back(str.substr(_find_offset, _split_offset - _find_offset));
141  _find_offset = _split_offset + delimiter.length();
142  }
143  _ret.push_back(str.substr(_find_offset, str.length() - _find_offset));
144  return _ret;
145 }
